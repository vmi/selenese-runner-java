FlowControl Test (from selenium-ide-flowcontrol)
open /form.html
store 10 x
gotolabel target1
runScript alert("This should not appear")
label target1
store 14 y
store 3 z
gotoIf ${y} > ${x} target2
runScript alert("This also should not appear");
label target2
echo "Compare : "+${x}+" vs "+${y}
while ${x} <= ${y}
storeEval ${x} + 1 x
echo Compare :  ${x}  vs  ${y}
while ${z} > 0
storeEval ${z} - 1 z
endWhile
endWhile
gotoLabel target3
label target3
echo Finished with x = ${x} and y = ${y}
